part 1

syms Ic Vbe Vt Ae q Dn ni Nb Wb;

Is = Ae * q * Dn * ni^2 / (Nb * Wb);
eq = Ic == Is * exp(Vbe / Vt);

solve(eq, Ae)

 
ans =
 
(Ic*Nb*Wb*exp(-Vbe/Vt))/(Dn*ni^2*q)

part 2

syms Vs(t) R C Vc(t);

eq = Vs(t) == R * C * diff(Vc, t) + Vc(t);
dsolve(eq)

 
ans =
 
C1*exp(-t/(C*R)) + exp(-t/(C*R))*int((exp(t/(C*R))*Vs(t))/(C*R), t, 'IgnoreSpecialCases', true, 'IgnoreAnalyticConstraints', true)
